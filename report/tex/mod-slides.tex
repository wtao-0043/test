\onslide*{2-3}{
\begin{proof}[Notes]
\begin{itemize}
\item<2-> 
Stick to a small set of \emph{conferences} and \emph{journals}:
	\begin{description}
	\item[Conferences]<2->
	No more than two in each tier:
	\begin{itemize}
	\item AAAI, IJCAI, UAI, ICLR, ICML, NeurIPS, KDD, CVPR, S\&P, NDSS
	\item ECAI, ACML, ECML/PKDD, ICDM, SDM, CIKM, ESORICS, ACSAC
	\item PAKDD, KSEM, PRICAI, IJCNN
	\end{itemize}
	
	\item[Journals]<2->
	No more than two in each tier:
	\begin{itemize}
	\item TIFS, PAMI, TPDS, AIJ, MLJ, JMLR, JAIR
	\item TKDE, TFS, TII, TNNLS, TC, PR
	\item EWSA, KAIS, FGCS
	\end{itemize}	
	\end{description}

\item<3->
Read some recent papers in the selected venues,
and be familiar with their writing \emph{style}, 
\emph{length} and paper \emph{structure}.
\end{itemize}
\end{proof}
}

\onslide*{4-}{
\begin{proof}[Notes]
\begin{itemize}
\item<4-> 
Never ever \emph{double submission}!
	\begin{itemize}
	\item<5->
	\emph{Double submission} may lead to \emph{double publication},
	and it is \emph{unethical}.
	
	\item<6->
	The same idea should not be published twice in journals.
	\end{itemize}
	
\item<7->
Research work published in conferences,
can be extended for journals.
\begin{itemize}
\item<8->
Typically the conference paper needs to be extended for \SI{30}{\percent}.
\end{itemize}

\item<9->
\faMagic { } \textsc{TULIP-Lab} suggested practise:
\begin{itemize}
\item<10->
Complete the full draft, 
and shrink down \SI{30}{\percent} for conference submission;

\item<11->
Once accepted,
submit the full version to journal.
\end{itemize}
\end{itemize}
\end{proof}
}

 				    \KwIn{database $D$, $\epsilon$, $\delta$,

					\KwOut{$\omega^*$}

					\For{$i \leftarrow 1$ \KwTo $T$}{
					$i_t \sim $ \UNI{$1, \cdots, n$}
					\tcp*[f]{select a random data example}\;
					$g_t = \bigtriangledown 
						l(\omega_{t-1}, (\vec{x}_{i_t}, y_{i_t})) 
						+ \lambda\bigtriangledown R(\omega_{t-1})$ \;
					$\omega_{t} = \prod_{\omega}(\omega_{t-1} - \eta_t g_t)$\;
					\tcp*[f]{projection onto the convex feasible set $\omega$}\;
					}
					
					\KwRet{$\omega^* = \omega_{T}$\;}
